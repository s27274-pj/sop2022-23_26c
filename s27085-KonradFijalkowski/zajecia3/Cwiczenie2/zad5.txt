#!/usr/bin/env bash
TABLICA=()
for i; do                     #bez dodawania 'in' pętla przeiteruje wszystkie argumenty
    TABLICA+=("$i")
done
for i in "${TABLICA[@]}"; do
    cat "$i"
done
echo ""
for i in "${TABLICA[@]}"; do
    tac "$i" | rev
done
