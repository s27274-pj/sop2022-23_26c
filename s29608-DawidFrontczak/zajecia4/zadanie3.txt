#include <ctype.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(int argc, char const *argv[]) {
    FILE *file;
    unsigned char bytes[16];
    int i;
    int n = 0;
    int odczyt = 0;

    if (argc == 1) {
		printf("Nie podano pliku\n");
		exit(1);
    } else {
        printf("Wczytywanie pliku %s\n", argv[1]);
        file = fopen(argv[1], "rb");
        if (file == NULL) {
            printf("BLAD\n");
            return 1;
        }
    }

    while ((odczyt = fread(&bytes, sizeof(char), 16, file)) > 0) {
        printf("%08X  ", n);
        for (i = 0; i < odczyt; i++) {
            printf("%02X ", bytes[i]);
        }
        for (i = 0; i < 16 - odczyt; i++) {
            printf("   ");
        }

        for (i = 0; i < odczyt; i++) {
            if (isprint(bytes[i])) {
                putchar(bytes[i]);
            } else if (bytes[i] == '\n') {
                putchar(' ');
            } else {
                printf("â–’");
            }
        }

        putchar('\n');
        n += 16;
        memset(bytes, 0, 16);
    }

    if (file != stdin) {
        printf("Zamkniecie pliku %s\n", argv[1]);
        fclose(file);
    }

    return 0;
}
