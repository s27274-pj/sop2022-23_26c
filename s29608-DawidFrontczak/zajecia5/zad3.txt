#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main(int argc, char* argv[]){
  int suma;
  int i;
  
  if (fork() > 0){
    wait(0);
    for(i=0; i<atoi(argv[1]); i++){
      suma += i;
    }
    printf("Wynik: %d\n", suma);
  }
  else{
    printf("Liczby nieparzyste: \n");
    for (i=0; i<=atoi(argv[1]); i++){
      if (i%2==1){
        printf("%d\n", i);
      }
    }
  }
  return 0;
}
